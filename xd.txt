    def buscar_nombre(self): # Busca por nombre al apretar la lupa
        global tree
        nombre_buscador = self.entry_buscador.get()

        if self.entry_buscador.get() == ' ' or self.entry_buscador.get() == '':
            print("Se escribio un caracter, no un producto")
            self.mostrar_inventario()
        else:
            self.cur.execute("SELECT * FROM inventario WHERE nombre=?", (nombre_buscador,))
            # Limpia el Treeview
            for item in self.tree.get_children():
                self.tree.delete(item)

            # Agrega los resultados de la consulta al Treeview
            for row in self.cur.fetchall():
                self.tree.insert('', tk.END, values=row)
